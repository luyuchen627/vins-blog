        # Redux

        Redux 由 Flux 演变而来，但受 Elm 的启发，避开了 Flux 的复杂性，Redux的理念在于提供可预测化的状态管理

        # 简单了解

        Redux应用的 state 被储存在唯一一个只读的 object tree 中，只能通过触发 action 来改变 state ， action 为改变的描述对象，具体改变需要通过编写 reducer 完成

        # 三大原则

        1. 单一数据源

        一个Redux应用只能有一个数据源，并且是只读的

        1. 单向数据流

        状态的修改只能通过触发 action 的方式来改变

        1. 使用纯函数改变状态

        reducer 函数应该是纯函数，不改变原状态，而是返回一个新状态

        有了这三个原则，可以使状态变得透明，可预测，可控制

        # 简单使用

        以用户信息的状态为例，简单了解Redux的使用

        # 定义action

        action 就是一个如何修改状态的描述的对象，有一个属性 type 辨识 action

        action可以是一个纯对象，也可以是一个函数，返回一个描述对象

        定义一个用户信息的更新action

        // actioner.js

// 用来创建action的函数
export default function actioner (type, payload) {
	return function (payload) {
		return {
			type,
			payload
		}
	}
}

        // actions.js

import actioner from './actioner'

// 显示声明action类型，使用Symbol类型，防止重复，也可使用别的类型
// 复杂项目，可以把action的类型单独声明在一个文件中，便于明确项目的action，减少样板代码
const USERINFO_UPDATE = Symbol('userinfo_update')

export {
	updateUserAction: actioner(USERINFO_UPDATE)
}

        # 定义reducer

        当触发action时，Redux会调用注册的reducer来具体变更状态，同时传入两个参数，state 和 action

        // reducer.js

// state为当前状态，action为触发的action，reducer应该是一个纯函数，并返回一个新状态
export default function reducer (state, action) {
	switch (action.type) {
		case USERINFO_UPDATE:
			return Object.assign({}, state, {
				user: action.payload
			})
		default:
			break;
	}
}

        # 定义Redux应用store

        Store 就是提供状态的，同时把 action 和 reducer 联系到一起的对象

        Store主要有以下几个功能：

        • 提供应用的 state
        • 提供 getState() 方法获取 state
        • 提供 dispatch(action) 方法触发 reducer 更新 state
        • 通过 subscribe(listener) 监听状态的变化
        • 通过 subscribe(listener) 函数返回的函数注销监听器

        相关的 createStore 的简单实现

        function createStore(reducer, initialState) {
	// 状态
	let state = initialState || {}
	
	// 监听器队列
	let listeners = []

	function getState () {
		return state
	}

	function dispatch (action) {
		try {
			// 触发reducer，获取新状态
			let newState = reducer(state, action)
			if (state !== newState) {
				// 更新状态
				state = newState
				// 触发监听
				listeners.map(i => { typeof i === 'function' && i() })
			}
		} catch (error) {
			throw error
		}
	}

	function subscribe (fn) {
		listeners.push(fn)

		// 返回一个取消监听器的函数
		return function unsubscribe() {
			listeners = []
		}
	}

	return {
		getState,
		dispatch,
		subscribe
	}
}

        # 异步的action

        异步的同步的区别就在于，触发action的时刻不可控制，我们可以用一个状态表示为来控制异步触发

        以一个异步请求为例

        // 开始请求，只定义类型，没有更改数据
const startFetch = {
	type: 'ASYNC_READY'
}

// 请求成功
const fetchSuccess = {
	type: 'ASYNC_SUCESS',
	payload: {}
}

// 请求失败
const fetchFail = {
	type: 'ASYNC_FAIL'
}

        我们可以在异步的不同时刻触发不同的action来异步更新状态，当然也可以触发同一个类型，用一个状态位表示是否更新

        # 中间件

        中间件就是根据 AOP 改变过来的，中间件可以帮助我们掌握状态更新前后的变化，我们可以相应的进行日志记录，回滚重演，持久化存储等等

        一个简单的中间件其实就是函数的劫持

        // 函数劫持
function hijack (fn, ns) {
	return function newFn() {
		console.log('函数执行前')
		let result = fn.apply(ns || null, arguments)
		console.log('函数执行后')
		return result
	}
}

        中间件的原理就是函数劫持，但是如果中间件很多，那么每一个中间件都对原函数封装一边，显得过于麻烦，我们采用参数注入的方式，把原函数传递进来

        1. 首先我们需要一个装载中间件的函数
        // 中间件的装载封装函数，ctx为当前上下文，middlewares为中间件列表
loadMiddleware(ctx, middlewares) {
	let index = -1  // 当前准备执行的middleware下标

	// 以递归调用的方式代替循环，好处是不需要额外变量即可获取上一个变化
	function dispatch(i) {
		if (i <= index) {
			// 当index改变时，实际调用的i小于等于index时，next会被重复调用，因此抛出错误
			throw new Error('next() called multiple times')
		}
		// 获取当前中间件
		let fn = middlewares[i]
		// 最后一个中间件判断
		if (i === middlewares.length) {
			return Promise.resolve()
		}
		if (!fn) {
			throw new Error('fn is not exist')
		}
		try {
			// 把上下文和上一个中间件(函数形式)以参数的方式注入下一个中间件
			// 使用promise可以支持异步的操作
			return Promise.resolve(fn(ctx, function next() {
				return dispatch(i + 1)
			}))
		} catch (err) {
			return Promise.reject(err)
		}
	}
	dispatch(0)
}
        1. 定义中间件
        // 定义一个对action类型检查的中间件
function checkType(ctx, next) {
	next()
	let dispatch = ctx.dispatch
	ctx.dispatch = function (action) {
		if (typeof action.type !== 'symbol') {
			console.log('action的类型请使用Symbol类型')
			return !1
		}
		dispatch.call(ctx, action)
	}
}

// 定义一个记录触发action的日志中间件
function logDispatch (ctx, next) {
	let dispatch = ctx.dispatch
	ctx.dispatch = function (action) {
		console.log('触发', action.type)
		console.log('触发前的状态为：', ctx.getState())
		dispatch.call(ctx, action)
		console.log('触发后的状态为：', ctx.getState())
	}
	next()
}
        1. 装载中间件
        // store为我们的Redux容器
loadMiddleware(store, [logDispatch, checkType])
