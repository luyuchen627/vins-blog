                    # 常用设计模式

                    # 单例模式

                    单例模式的关键是内部有一个变量指向一个实例

                    var SingletonCreater = (function () {  // SingletonCreater工厂
	// 实例构造器
    function Singleton(args) {
        var args = args || {}
        this.name = args.name
    }
	// 实例容器
    var instance
	// 暴露公开的方法
    var _static = {
        getInstance: function (args) {
            if (instance === undefined) {
                instance = new Singleton(args)
            }
            return instance
        }
    }
    return _static
})()
// 创建单例
var singletonOb = SingletonCreater.getInstance({ name: 'vins' })
console.log(singletonOb.name) // 输出 vins

                    使用场景：

                    • 资源共享的情况下，避免由于资源操作时导致的性能或损耗等。如日志文件，应用配置
                    • 资源控制的情况下，方便资源之间的互相通信。如XMLHttpRequest, 数据库连接

                    # 工厂模式

                    工厂模式可以根据具体环境创建不同实例

                    // animal类
var animal = function () {
	this.say = function () {}
}

// dog
animal.dog = function () {
	// 重写
	this.say = function () {
		console.log('汪汪')
	}
}

// bird
animal.bird = function () {
	// 重写
	this.say = function () {
		console.log('叽叽喳喳')
	}
}

var animalfactory = function (type) {
	if (!animal[type]) {
		throw new Error('类型不存在')
	}
	return new animal[type]()
}
// 创建动物为dog的工厂
var animal1 = animalfactory('dog')
animal1.say() // 输出 汪汪

                    使用场景：

                    • 创建大量的统一对象
                    • 根据具体的行为创建对象（多态）

                    # 装饰器模式

                    装饰者模式是为已有功能动态地添加更多功能的一种方式，优点是把类（函数）的核心职责和装饰功能区分开了。js使用高阶函数实现

                    var bird = function () {}

bird.prototype.say = function () {
	console.log('叽叽喳喳')
}

function decorate (target) {
	var say = target.prototype.say
	target.prototype.say = function () {
		var _self = this
		console.log('汪汪')
		say.call(_self)
	}
	return target
}
decorate(bird)
var bird1 = new bird()
bird1.say()  // 输出 汪汪 叽叽喳喳

                    使用场景：

                    • 对原有对象的扩展
                    • 统一对象的行为，如性能监控，日志管理修饰

                    # 代理模式

                    代理模式是为其他对象提供一种代理以控制对这个对象的访问。

                    var person = function () {
	this.meet = function () {}
}
var star = function () {}

var starProxy = function (star) {
	this.star = star
	this.meet = function () {
		(new person()).meet()
	}
}
// 明星代理粉丝约见明星
(new starProxy()).meet()

                    当我们需要使用的对象很复杂或者需要很长时间去构造，这时就可以使用代理模式 使用场景：

                    • 一个对象，比如一幅很大的图像，需要载入的时间很长。
                    • 一个需要很长时间才可以完成的计算结果，并且需要在它计算过程中显示中间结果
                    • 访问权限控制

                    # 观察者模式

                    当一个对象的改变需要同时改变其它对象，并且它不知道具体有多少对象需要改变的时候，就应该考虑使用观察者模式

                    // 定义观察类
var observer = function (nameSpace) {
	// 订阅者容器
	this.subscribers = nameSpace || {}

    // 订阅（可以订阅多个）
    this.addSubscriber = function (name, cb) {
		if (!this.subscribers[name]) {
			this.subscribers[name] = []
		}
        this.subscribers[name].push(cb)
    }
    // 退订
    this.removeSubscriber = function (name) {
        if (this.subscribers[name]) {
			delete this.subscribers[name]
		}
    }
    // 发布
    this.publish = function (name, msg) {
        for (var i = 0; i < this.subscribers[name].length; i++) {
            if (typeof this.subscribers[name][i] === 'function') {
                this.subscribers[name][i](msg)
            }
        }
    }
	// 创建观察者
	createObserver (obj) {
		for (var k in this) {
			obj[k] = this[k]
			obj.subscribers = {}
		}
	}
}

// 创建观察者
var observer1 = new observer()
// 订阅事件
observer1.addSubscriber('log', console.log)
observer1.publish('log', 'hello')

                    使用场景：

                    • dom事件订阅
                    • 数据变化
                    • 消息广播

                    # 策略模式

                    策略模式就是定义一系列的算法(策略)，把他们一个个封装起来，并且是他们可以相互替换

                    // 定义一个根据用户的等级获取商品价钱的类
var good = function (price) {
	this.price = price
	this.priceRule = {}
	this.getPrice = function(user) {
		return this.priceRule[user.level] ? this.priceRule[user.level].call(this) : this.price
	}
}
// 定义用户类
var user = function (level) {
	this.level = level
	this.buyGood = function (good) {
		console.log(good.getPrice(this))
	}
}

// 构建商品实例
var good1 = new good(100)
// 扩展价格策略
good1.priceRule.level1 = function () {
	return this.price * 0.9
}
good1.priceRule.level2 = function () {
	return this.price * 0.8
}
good1.priceRule.level3 = function () {
	return this.price * 0.6
}
// 构建实例
var user1 = new user('level2')
user1.buyGood(good1)  // 输出 80

                    使用场景：

                    • 某一个功能有多种独立方案可以选择

                    # 迭代器模式

                    迭代器模式是使用提供一种方法可以遍历整个对象，同时不暴露对象内部表示

                    function createIterator (items) {
    var i = 0
    return {
        next: function () {
            var done = (i >= items.length)
            var value = !done ? items[i++] : undefined
            return {
                done: done,
                value: value
            }
        }
    }
}
var iterator = createIterator([1, 2, 3])
console.log(iterator.next()) // "{ value: 1, done: false }"
console.log(iterator.next()) // "{ value: 2, done: false }"
console.log(iterator.next()) // "{ value: 3, done: false }"
console.log(iterator.next()) // "{ value: undefined, done: true }"
// 之后的所有调用
console.log(iterator.next()) // "{ value: undefined, done: true }"

                    使用场景：

                    • 缓存计算，使用时输出
                    • 提供可遍历的方法

                    # 中介者模式

                    中介者模式就是在多个之间设置一个中介者，用于维护它们相互之间的通讯和委托

                    var user = function (name) {
	this.name = name
	this.play = function () {
		return Math.random()
	}
}
var mediator = function () {
	this.users = []
	this.show = function (res) {
		var sortedRes = res.sort(function (a, b) {
			return a.num - b.num
		})
		console.log(sortedRes)
	}
	this.addUser = function (user) {
		this.users.push(user)
		return this
	}
	this.begin = function () {
		var result = this.users.map(function (u) {
			return {
				name: u.name,
				num: u.play(),
			}
		})
		this.show(result)
	}
}
var user1 = new user('user1')
var user2 = new user('user2')
var user3 = new user('user3')
var mediator1 = new mediator()
mediator1.addUser(user1).addUser(user2).addUser(user3)
mediator1.begin()

                    使用场景：

                    • 一组对象已定义良好但是以复杂的方式进行通信的场合
                    • 作为观察者模式上的消息Bus

                    # 状态模式

                    使用场景：

                    • 一个对象的行为取决于它的状态，并且它必须在运行时刻根据状态改变它的行为。
                    • 一个操作中含有大量的分支语句，而且这些分支语句依赖于该对象的状态。
