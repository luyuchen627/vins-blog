                                          # 原型

                                          # 一些概念

                                          1. 原型对象 所有实例继承和共享的对象 原型对象也是对象

                                          2. 构造函数 用于生成实例的函数 构造函数的 prototype 指向原型对象

                                          3. 实例 通过new出来的对象 实例有一个 **__proto__**或者 [[Prototype]] 属性指向原型对象 实例的 constructor 属性指向构造器

                                          4. 原型链 JavaScript 对象有一个指向一个原型对象的链。 当试图访问一个对象的属性时，它不仅仅在该对象上搜寻，还会搜寻该对象的原型，以及该对象的原型的原型，依次层层向上搜索，直到找到一个名字匹配的属性或到达原型链的末尾。

                                          # 实例详解

                                          // 定义一个对象(类，或者构造器)
let f = function () {
   this.a = 1
   this.b = 2
}

// 初始化一个实例
let o = new f() // {a: 1, b: 2}

// 在原型对象上添加属性和方法
f.prototype.a = 'a'
f.prototype.person = [ 'vins' ]
f.prototype.getPerson = function () {
	return this.person
}

// 注意：不要在原型对象上直接定义 f.prototype = { a: 1, b: 2 } 这样是原型的重写，会直接打破原型链

                                          # 原型分析

                                          1. 概念：

                                            • 实例: o
                                            • 原型对象: f.prototype
                                            • 构造器: f

                                          2. 关系:

                                            • o.__proto__ === o.[[Prototype]] === f.prototype
                                            • o.__proto__.constructor === f

                                          3. 原型连

                                            • o.__proto__.__proto__ === Object.prototype
                                            • o.__proto__.__proto__.__proto__ === null // 原型链的尽头是null

                                          4. 整个原型链

                                            • o -> f.prototype -> Object.prototype -> null

                                          # 原型修改

                                          1. 直接修改 __proto__
                                          2. 通过 Object.setPrototypeOf 设置原型，另外通过 Object.getPrototypeOf 获取原型

                                          # 实例属性访问

                                          1. 如果实例有这个属性，返回实例的属性值
                                          2. 否则沿着原型链向上寻找这个属性，直到原型链的末尾, 存在则返回，不存在则为undefined

                                          # 属性的遍历

                                          • 在原型链上查找属性比较耗时，对性能有副作用。另外，试图访问不存在的属性时会遍历整个原型链
                                          • 遍历对象的属性时，原型链上的每个可枚举属性都会被枚举出来。

                                          各种遍历方法：

                                          1. for...in循环：只遍历对象自身的和继承的可枚举的属性

                                          2. Object.keys()：返回对象自身的所有可枚举的属性的键名

                                          3. JSON.stringify()：只格式化对象自身的可枚举的属性

                                          4. Object.assign()：只合并对象自身的可枚举的属性

                                          5. Object.getOwnPropertyNames(obj): 获取除Symbols属性外的所有属性的键名

                                          6. Object.getOwnPropertySymbols(obj): 获取所有Symbols属性的键名

                                          7. Reflect.ownKeys(obj): 获取所有属性的键名

                                          要检查对象是否具有自己定义的属性，而不是其原型链上的某个属性；可以使用从Object.prototype继承的 hasOwnProperty 方法，或者使用 Object.keys() 遍历

                                          # 对象的创建方式

                                          # 字面量

                                          var o = {
	a: 1
}

// 原型链 o -> Object.prototype -> null

                                          # new 操作符

                                          var o = new Object({
	a: 1
})

// 原型链 o -> Object.prototype -> null

                                          # new 操作符原理

                                          function new (F, ...arg) {
	// 1. 创建一个空对象
    var obj = {}
	// 2. 设置'__proto__'属性指向原型
	obj['__proto__'] = F.prototype
	// 3. 使用构造器实例化，并修改this的指向
	var res = F.apply(obj, arg)
	// 4. 返回对象(如果构造器有返回并且返回的类型为object，则返回构造器的返回数据，否则返回实例化后的对象)
	return typeof res === 'object' ? res : obj
}

                                          初始化函数调用的是类函数本身，和构造器(constructor)没有关系;举例，当改写原型对象时，构造器也会随之改变，但是并不影响实例的初始化

                                          // 父类
function Parent (name) {
	this.name = name || 'parent'
}
Parent.prototype.sayName = function () {
	return this.name
}

// 子类
function Child (name) {
	this.childName = name || 'child'
}
// 改写原型，此时Child.prototype.constructor === Parent
Child.prototype = new Parent()

// 实例化
var o = new Child()
console.log(o.constructor)  // Parent
console.log(o)  // Child {childName: "child"}；依旧使用Child函数实例化对象

                                          # super 关键字

                                          1. 在构造器中使用，相当于Parent.prototype.constructor.call(this)
                                          2. 在类方法中使用，相当于Parent.prototype.methos.call(this)

                                          # class 简单创建原理

                                          // class构造函数
function creatClass (Constructor, protoProps, staticProps) {
    // 方法添加到原型上
    if (protoProps) {
		defineProperties(Constructor.prototype, protoProps)
	}
	// 静态方法添加到类上
	if (staticProps) {
		defineProperties(Constructor,staticProps)
	}
	return Constructor
}

// 定义对象属性
function defineProperties (target, props) {
	for (var i = 0; i < props.length; i++) {
       var descriptor = props[i]
       Object.defineProperty(target, descriptor.key, descriptor)
	}
})

                                          # 继承

                                          // 父类
function Parent (name) {
	this.name = name || 'parent'
}
Parent.prototype.sayName = function () {
	return this.name
}

                                          # 原型继承

                                          // 子类
function Child (name) {
	this.childName = name || 'child'
}
// 子类原型重写
Child.prototype = Parent.prototype
// 修改子类的构造器指向
Child.prototype.constructor = Child

var o = new Child()
console.log(o.sayName())  // undefined

                                          实现：该方式继承主要是使子类的原型为父类的原型 优点：只能继承父构造函数的原型对象上的成员, 不能继承父构造函数的实例对象的成员 缺点：原型对象的成员共享问题，不能向父类构造器传参

                                          # 原型链继承

                                          // 子类
function Child (name) {
	this.childName = name || 'child'
}
// 子类原型链继承
Child.prototype = new Parent()
// 修改子类的构造器指向
Child.prototype.constructor = Child

var o = new Child()
console.log(o.sayName())  // 'parent'

                                          实现：该方式继承主要是修改子类的原型 优点：会继承子类，父类和父类实例的所有成员 缺点：原型对象的成员共享问题，不能向父类构造器传参

                                          # 组合继承

                                          // 子类
function Child (name) {
	// 调用父类的构造函数,同时向父类构造器传参
	Parent.call(this, name)
	this.childName = name || 'child'
}
// 设置原型继承
Child.prototype = Parent.prototype
// 修改子类的构造器指向
Child.prototype.constructor = Child

var o = new Child('child')
console.log(o.sayName())  // 'child'

                                          实现：该方式继承主要是主动调用父类的构造函数，同时设置原型 优点：会继承子类，父类的所有成员，不能继承父构造函数的实例对象的成员，父类实例对象其实已经都在子类实例对象初始化了 缺点：原型对象的成员共享问题

                                          # 扩展继承

                                          // 子类
function Child (name) {
	// 调用父类的构造函数,同时向父类构造器传参
	Parent.call(this, name)
	this.childName = name || 'child'
}
// 使用深拷贝实现继承
deepCopy(Child.prototype, Parent.prototype)
// 修改子类的构造器指向
Child.prototype.constructor = Child

var o = new Child('child')
console.log(o.sayName())  // 'child'

                                          实现：该方式继承主要是主动调用父类的构造函数，同时把父类原型上的所有成员深复制到子类原型 优点：解决原型对象的成员共享问题；可以实现多继承；会继承子类，父类的所有成员

                                          # 匿名对象继承函数封装

                                          function inherit(proto) {
    if (proto == null) throw TypeError()
	// 如果Object.create()存在,则使用
    if (Object.create) return Object.create(proto)

	// 类型检测
    var t = typeof proto
    if (t!=='object' && t!=='function') throw TypeError()
	// 定义一个空构造函数
    var F = function() {}
    F.prototype = proto
	// 返回一个继承了proto的实例
    return new F()
}

                                          # 代码风格

                                          • 类定义时，首字母大写
                                          • 类定义时，禁止被作为方法调用
                                          • 保护私有属性，只提供读写私有属性的方法
                                          • 原型上只定义方法，不要定义属性，如果定义，请注意属性共享问题
