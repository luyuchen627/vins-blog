            # 闭包

            # 概念

            通俗的理解：闭包是指有权访问另一个函数作用域中的变量的函数 mdn的解释：闭包是函数和声明该函数的词法环境的组合，这个环境包含了这个闭包创建时所能访问的所有局部变量

            # 优势

            1. 创建独立作用域，约束变量，保存变量引用
            2. 提供内部变量的访问权限

            # 劣势

            闭包在处理速度和内存消耗方面对脚本性能具有负面影响

            # 应用

            1. 内部缓存数据
            // 记忆函数
function memory(fn) {
	var r = {}
	return function(n) {
		if (!r[n]) {
			r[n] = fn(n)
		}
		return r[n]
	}
}

// 斐波那契数列生成函数
function fibFn (n) {
	return n > 1 ? fibFn(n-1) + fibFn(n-2) : Math.max(n, 0)
}
// 斐波那契数列生成函数(另一种形式)
function fibFn (n, a = 0, b = 1) {
	return n > 1 ? fibFn (n - 1, b, a + b) : b
}

var fib = memory(fibFn)
fib(-2) // 0
fib(1)  // 1
fib(5)  // 5
            1. 单例判断
            var SingletonCreater = (function () {  // SingletonCreater工厂
	// 实例构造器
    function Singleton(args) {
        var args = args || {}
        this.name = args.name
    }
	// 实例容器
    var instance
	// 暴露公开的方法
    return {
        getInstance: function (args) {
            if (instance === undefined) {
                instance = new Singleton(args)
            }
            return instance
        }
    }
})()
// 创建单例
var singletonOb = SingletonCreater.getInstance({ name: 'vins' })
console.log(singletonOb.name) // 输出 vins
            1. 控制属性的访问
            function bird () {
	var name = 'vins'
	return {
		setName (n) {
			name = n
		},
		getName () {
			return name
		}
	}
}

var birdAccessor = bird()
birdAccessor.getName()       // 'vins'
birdAccessor.setName('aha')
birdAccessor.getName()       // 'aha'
            1. 偏函数
            function add(x) {
	return function(y) {
		return x + y
	}
}
var add5 = add(5)
var add10 = add(10)
add5(1)   // 6
add10(1)  // 11

            # 注意点

            1. 在循环中创建闭包，引用的变量可能发生变化
            for (var i = 0; i < 5; i++ ) {
	setTimeout(function () {
		console.log(i)
	}, i * 1000)
}
// 上述情况会依次打印5个5
// 改进,再包裹一层匿名闭包，包含i变量的引用
for (var i = 0; i < 5; i++ ) {
	(function(i) {
		setTimeout(function () {
			console.log(i)
		}, i * 1000)
	})(i)
}
// 依次打印0,1,2,3,4
            1. this的指向不明
            var obj = {
	name: 'obj',
	getName: function() {
        return function() {
			console.info(this.name)
        }
    }
}

obj.getName()()  // undefined, 因为此时this指向window

// 改进，闭包引起的this指向问题，关键点需要保存this的指向，或动态改变this的指向
var obj = {
	name: 'obj',
	getName: function() {
		var _self = this  // 保存this的指向
        return function() {
			console.info(_self.name)
        }
    }
}

obj.getName()()  // 'obj'
