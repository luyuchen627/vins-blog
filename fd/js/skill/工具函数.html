# 睡眠

// 全局睡眠，很影响性能
function sleep (t) {
	var now =  Date.now()
	while (Date.now() - now < t) {}
}

// 基于promise的异步睡眠

function sleep (t) {
	return new Promise((resolve, reject) => {
		setTimeout(() => {
			resolve()
		}, t)
	})
}

# 事件绑定

const on = (function () {
	if (document.addEventListener) {
		return function (element, event, handler) {
			if (element && event && handler) {
				element.addEventListener(event, handler, false)
			}
		}
	} else {
		return function (element, event, handler) {
			if (element && event && handler) {
				element.attachEvent('on' + event, handler)
			}
		}
	}
})()

# 事件解绑

const off = (function () {
	if (document.removeEventListener) {
		return function (element, event, handler) {
			if (element && event) {
				element.removeEventListener(event, handler, false)
			}
		}
	} else {
		return function (element, event, handler) {
			if (element && event) {
				element.detachEvent('on' + event, handler)
			}
		}
	}
})()

# localStorage

const localData = {
	set(key, value) {
		let v = null
		if (typeof value === 'object') {
			try {
				v = JSON.stringify(value)
			} catch (e) {
				console.log(v)
			}
		} else {
			v = value
		}
		localStorage.setItem(key, v)
	},
	get(key, defaultVal) {
		let v = localStorage.getItem(key)
		if (typeof v === 'string') {
			try {
				if (v.search(/^\{/) > -1 || v.search(/^\[/) > -1) {
					return JSON.parse(v)
				} else {
					return v
				}
			} catch (e) {
				console.log(e, v)
				return defaultVal
			}
		}
		return v
	},
	remove(key) {
		localStorage.removeItem(key)
	},
	clear() {
		localStorage.clear()
	}
}

# url编码

function urlFormat(obj, encodeType) {
	let urlArray = []
	let urlStr
	for (let key in obj) {
		let val = typeof obj[key] === 'object' ? JSON.stringify(obj[key]) : obj[key]
		urlArray.push(`${key}=${val}`)
	}
	urlStr = urlArray.join('&')
	return encodeType === 'URI' ? encodeURI(urlStr) : urlStr
}

# url解码

function urlParse(url, encodeType) {
	url = encodeType === 'URI' ? decodeURI(url) : url
	let urlObj = {}
	let str = url.split('?')[1]
	if (str) {
		let items = str.split('&')

		for (let i = 0, len = items.length; i < len; i++) {
			let row = items[i].split('=')
			urlObj[row[0]] = decodeURIComponent(row[1])
		}
	}
	return urlObj
}

# jsonParse

function parseJson(str) {
	if (typeof str === 'string') {
		try {
			if (str.search(/^\{/) > -1 || str.search(/^\[/) > -1) {
				return JSON.parse(str)
			} else {
				return str
			}
		} catch (e) {
			throw new Error('Cannot be converted to json format.')
		}
	} else {
		return str
	}
}

# 添加样式

function addStyle(customStyle, eleId) {
	let eleStyle = eleId ? document.getElementById(eleId) : false

	if (eleStyle) {
		eleStyle.innerHTML = customStyle
	} else {
		let newStyleEle = document.createElement('style')
		newStyleEle.type = 'text/css'
		newStyleEle.id = eleId
		newStyleEle.innerHTML = customStyle
		document.getElementsByTagName('head').item(0).appendChild(newStyleEle)
	}
}

# 重复点击

function checkRepeatClick() {
	let tapTime = 0
	let interval = 1000
	return (customInterval = interval, repeat = true) => {
		let nowTime = Date.now()
		let isRepeat = false
		if (nowTime - tapTime < customInterval) {
			isRepeat = !0
			if (!repeat) {
				tapTime = nowTime
			}
		} else {
			tapTime = nowTime
		}
		return isRepeat
	}
}

# formatDate

/*
** 时间戳转换成指定格式日期
** eg.
** dateFormat(11111111111111, 'Y年m月d日 H时i分')
** → "2322年02月06日 03时45分"
*/
function dateFormat(timestamp, formats) {
    // formats格式包括
    // 1. Y-m-d
    // 2. Y-m-d H:i:s
    // 3. Y年m月d日
    // 4. Y年m月d日 H时i分
    formats = formats || 'Y-m-d'

    var zero = function (value) {
        if (value < 10) {
            return '0' + value
        }
        return value
    }

    var myDate = timestamp? new Date(timestamp): new Date()

    var year = myDate.getFullYear()
    var month = zero(myDate.getMonth() + 1)
    var day = zero(myDate.getDate())

    var hour = zero(myDate.getHours())
    var minite = zero(myDate.getMinutes())
    var second = zero(myDate.getSeconds())

    return formats.replace(/Y|m|d|H|i|s/ig, function (matches) {
        return ({
            Y: year,
            m: month,
            d: day,
            H: hour,
            i: minite,
            s: second
        })[matches]
    })
}

# 过滤null和undefined

function filterNull (o) {
	const obj = {}
	for (const k in o) {
		if (o[k] != null) {
			obj[k] = o[k]
		}
	}
	return obj
}

# 类型判断

function typeis(obj, type) {
	if (!type) {
		throw new Error('Miss required parameter -- type')
	}
	type = type.toString().toLowerCase()
	type = type.substr(0, 1).toUpperCase() + type.substring(1)
		// var types = ["Array", "Boolean", "Date", "Number", "Object", "RegExp", "String", "Window", "HTMLDocument"]
	return Object.prototype.toString.call(obj) === '[object ' + type + ']'
}

function isMobile () {
	if (navigator) {
		return /Android|webOS|iPhone|iPod|BlackBerry/i.test(navigator.userAgent)
	}
	if (window || screen) {
		return window.innerWidth < 450 || screen.availWidth < 450
	}
	return !1
}
