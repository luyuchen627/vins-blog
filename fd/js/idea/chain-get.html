# 链式读取对象属性

# 属性的链式读取

对于一个深层次的结构，如果想要安全的读取某一级的属性，就要判断该级上一级的属性是否存在，例如：

// 对于对象obj

var text = obj.data && obj.data.body && obj.data.body.list && obj.data.body.list[0].text

多层次的属性读取的难以控制的，常用的解决方法就是通过短路运算来判断；JS的提案中的 链判断运算符可以解决这个痛点

# 链判断运算符简单使用

// 使用 ?. 操作符获取属性，如果不存在，则中断，返回undefined
var text = obj?.data?.body?.list[0]?.text

// babel 转码后的代码
var _obj$data, _obj$data$body, _obj$data$body$list$;

var obj = {};
var text = obj === null || obj === void 0 
	? void 0 
	: (_obj$data = obj.data) === null || _obj$data === void 0 
		? void 0 
		: (_obj$data$body = _obj$data.body) === null || _obj$data$body === void 0 
			? void 0 
			: (_obj$data$body$list$ = _obj$data$body.list[0]) === null || _obj$data$body$list$ === void 0 
				? void 0 
				: _obj$data$body$list$.text;

# 使用方法代替

简单思路: 在 Object 原型上添加一个获取属性的方法，如果属性存在，返回一个包含该方法的属性以链式调用，同时把属性值保存，否则，返回一个对象(为了链式调用)或者默认值

/**
 * @params k 属性值
 * @params def 默认值
 * 
 * @return 存在属性，返回一个对象，通过该对象的 v 属性获取真实的属性值, 否则返回一个默认值
 **/
var _g = function(k, def) {
	return this[k] ? 
		{ 
			// 保证链式调用的上下文（这里主要是针对原型上的属性）
			_g: _g.bind(this[k]),
			// 防止原型上的方法的上下文的混乱
			v: this.hasOwnProperty(k) ? 
				this[k] : 
				(typeof this[k] === 'function' ? this[k].bind(this) : this[k])
		} : 
		(this[k] = { v: def || undefined })
}

Object.prototype._g = _g

测试

var a = {b: 1, c: { d: 2}}

// 简单获取
a._g('b').v   // 1

// 获取原型上的方法
a._g('b')._g('toString').v()  // '1'

// 链式获取
a._g('c')._g('d').v   // 2

// 获取不存在的属性
a._g('e')._g('f', 'vins')  // 'vins'

// 获取不存在的属性
a._g('m')._g('n').v  // undefined

# 使用proxy代理

var chainGetProxy = function (obj, def) {
	return new Proxy(obj, {
		get (target, k) {
			return k === '_v' 
					? target[k] 
					: target[k] && typeof target[k] === 'object'
						? chainGetProxy(target[k])
						: chainGetProxy(Object.assign(Object.create(null), { _v: typeof target[k] === 'function' ? target[k].bind(this) : (target[k] || def) }))
		}
	})
}

示例：

var a = {a: 1, b: { log: function () { console.log(this.a) }}}
var b = chainGetProxy(a)

b.b.log._v  // function

b.a.b.c.d._v  // undefined

使用的时候，不管属性存在不存在，都要使用 _v 来获取真实值

# 属性的链式设置

对于一个深层次的对象，如果一个属性不存在，那么就没有办法给它添加属性，即给 undefined 添加属性会报错的

var a = {}

// 下面代码会报错
a.b.c.d = 1

以下是常见的几种方案

# 逐级添加

((a.b || a.b = {}).c || a.b.c = {}).d = 1

# 正则匹配

a.set('b.c.d') = 1

# 链式添加

a.set('b').set('c').set('d') = 1

方法实现

/**
 * @params k 键值
 * @params v 属性值，如果有属性值，返回this，否则返回this[k]
 * 
 * @return 存在属性，返回一个对象，通过该对象的 v 属性获取真实的属性值, 否则返回一个默认值
 **/
var _s = function(k, v) {
	return this[k] ? 
		( v ? ((this[k] = v) && this) : this[k]) : 
		((this[k] = (v !== undefined) ? v : {}) && this[k])
}

Object.prototype._s= _s
