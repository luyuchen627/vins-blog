            # sequelizejs-模型

            # 模型定义

            模型定义使用define函数

            # 简单定义

            // 简单定义
const Project = sequelize.define('project', {
	title: Sequelize.STRING,
	description: Sequelize.TEXT
})

            # 类型定义

            const Foo = sequelize.define('foo', {
	title: {
		type: Sequelize.String,       // 字段类型
		validate：{},                 // 验证
		allowNull: false,             // 允许为空
		defaultValue: true,           // 默认值
		primaryKey: true,             // 主键
		autoIncrement: true,          // 自动增加
		unique: true
	}
})

            # 字段的getter和setter的定义

            表的定义可以使用es5中的get和set，在查询的时候使用get返回，更新的时候使用set更新

            // 字段的getter和setter
const Foo = sequelize.define('foo', {
  name: {
    type: Sequelize.STRING,
    allowNull: false,
    get() {
      const title = this.getDataValue('title');
      return this.getDataValue('name') + ' (' + title + ')';
    },
  },
  title: {
    type: Sequelize.STRING,
    allowNull: false,
    set(val) {
      this.setDataValue('title', val.toUpperCase());
    }
  }
});

            # 模型选项定义

            // 时间戳处理
const Bar = sequelize.define('bar', { /* bla */ }, {

	timestamps: true,  // 时间戳处理(updatedAt, createdAt)，true添加，false不添加，默认添加

	createdAt: false,  // 记录创建时间戳，（true添加，false不添加）在timestamps为true时生效

	updatedAt: 'updateTimestamp', // 记录修改时间戳，（true添加，false不添加，字符串修改为别名）在timestamps为true时生效

	deletedAt: 'destroyTime', // 记录删除时间戳

	paranoid: true,  // 记录删除时，不要删除数据库条目，而是将新添加的属性deletedAt设置为当前日期，只有在timestamps开启时有效

	underscored: true,  // 使用下划线的形势添加字段名，默认使用camelcase方式，设置为true后，updatedAt会变为updated_at

	freezeTableName: true,  // 禁止修改表名

	tableName: 'custom_table_name',  // 自定义表名

	version: true,  // 开启版本锁定，实例过期会抛出错误
})

            # 模型作用域定义

            作用域有默认作用域和扩展作用域，默认的作用域定义在defaultScope，扩展的以对象的形势定义在scopes中

            // 作用域定义
const Project = sequelize.define('project', { // Attributes }, {
	defaultScope: {
		where: {
			active: true
		}
	},
	scopes: {
		custom: {
			include: [
				{ model: User, where: { active: true }}
			]
			where: {
				deleted: true
			},
			limit: 10,
			offset: 10
		},
	}
});
            • 默认作用域中添加的条件，关联等都会应用到该模型所有的操作
            • 通过Model.scope来调用自定义作用域，返回该模型的一个视图
            • Model.scope函数传递多个作用域名称，或者传递一个数组，可以合并作用域
            • 作用域合并，相同的条件，后者覆盖前者，操作时显视声明的会覆盖作用域

            # 钩子定义

            // 钩子定义
const User = sequelize.define('user', { // Attributes }, {
	hooks: {
		beforeValidate: (user, options) => {
			user.mood = 'happy'
		},
		afterValidate: (user, options) => {
			user.username = 'Toni'
		}
	}
})

            通过Model.hook和Model.addHook，或者直接添加hook的名称也可以添加钩子

            # 字段类型

            // 字段类型说明
	STRING	函数（长度= 255：整数）	可变长度的字符串

	CHAR	函数（长度= 255：整数）	固定长度的字符串。

	TEXT	函数（长度：字符串）	无限长度的文本列。可用长度：tiny，medium，long

	TINYINT	函数（长度：整数）	一个8位整数。

	SMALLINT	函数（长度：整数）	一个16位整数。

	MEDIUMINT	函数（长度：整数）	一个24位整数。

	INTEGER	函数（长度= 255：整数）	一个32位整数。

	BIGINT	函数（长度：整数）	64位整数。注意：定义为的属性BIGINT将被视为来自node-postgres的string此特性，以防止精度损失。要将此属性设置为a number，这是一种可能的解决方法。

	FLOAT	函数（长度：整数，小数：整数）	浮点数（4字节精度）。

	DOUBLE	函数（长度：整数，小数：整数）	浮点数（8字节精度）。

	DECIMAL	函数（精度：整数，小数：整数）	十进制数。

	REAL	函数（长度：整数，小数：整数）	浮点数（4字节精度）。

	BOOLEAN	函数	布尔/ tinyint列，取决于方言

	BLOB	函数（长度：字符串）	二进制存储。可用长度：tiny，medium，long

	ENUM	function （values：string []）	枚举。DataTypes.ENUM('value', 'another value')。

	DATE	函数（长度：整数）	日期时间列

	DATEONLY	功能	仅限日期列（无时间戳）

	TIME	功能	时间栏

	NOW	    功能	当前时间戳的默认值

	UUID	功能	存储唯一通用标识符的列。与UUIDV1或一起UUIDV4使用默认值。

	UUIDV1	功能	根据UUID v1标准生成的默认唯一通用标识符

	UUIDV4	功能	根据UUID v4标准生成的默认唯一通用标识符

	HSTORE	功能	键/值存储列。仅适用于Postgres。

	JSON	功能	JSON字符串列。可在MySQL，Postgres和SQLite中使用

	JSONB	功能	二进制存储JSON列。仅适用于Postgres。

	ARRAY	function （类型：DataTypes）	type例如，一个数组DataTypes.ARRAY(DataTypes.DECIMAL)。仅适用于Postgres。

	范围	function （类型：DataTypes）	范围类型是表示某种元素类型的值范围的数据类型（称为范围的子类型）。仅适用于Postgres。有关更多详细信息，请参阅Postgres文档

	GEOMETRY	function （type：string，srid：string）	存储几何信息的列。它仅适用于PostgreSQL（使用PostGIS）或MySQL。在MySQL中，允许的几何类型POINT，LINESTRING，POLYGON。

            提示：除了上面提到的类型之外，integer，bigint，float和double还支持unsigned和zerofill属性，它们可以按任何顺序组合：请注意，这不适用于PostgreSQL！

            	Sequelize.INTEGER.UNSIGNED              // INTEGER UNSIGNED
	Sequelize.INTEGER(11).UNSIGNED          // INTEGER(11) UNSIGNED
	Sequelize.INTEGER(11).ZEROFILL          // INTEGER(11) ZEROFILL
	Sequelize.INTEGER(11).ZEROFILL.UNSIGNED // INTEGER(11) UNSIGNED ZEROFILL
	Sequelize.INTEGER(11).UNSIGNED.ZEROFILL // INTEGER(11) UNSIGNED ZEROFILL

            # 字段验证

            • is: ["^[a-z]+$",'i'] or /^[a-z]+$/i, // 正则
            • not: ["[a-z]",'i'], // 否定匹配
            • isEmail: true, // 邮箱检测
            • isUrl: true, // 地址检测
            • isIP: true, // 是否是ip地址，ipv4或者ipv6
            • isIPv4: true, // checks for IPv4 (129.89.23.1)
            • isIPv6: true, // checks for IPv6 format
            • isAlpha: true, // will only allow letters
            • isAlphanumeric: true, // will only allow alphanumeric characters, so "_abc" will fail
            • isNumeric: true, // will only allow numbers
            • isInt: true, // checks for valid integers
            • isFloat: true, // checks for valid floating point numbers
            • isDecimal: true, // checks for any numbers
            • isLowercase: true, // checks for lowercase
            • isUppercase: true, // checks for uppercase
            • notNull: true, // won't allow null
            • isNull: true, // only allows null
            • notEmpty: true, // don't allow empty strings
            • equals: 'specific value', // only allow a specific value
            • contains: 'foo', // force specific substrings
            • notIn: [['foo', 'bar']], // check the value is not one of these
            • isIn: [['foo', 'bar']], // check the value is one of these
            • notContains: 'bar', // don't allow specific substrings
            • len: [2,10], // only allow values with length between 2 and 10
            • isUUID: 4, // only allow uuids
            • isDate: true, // only allow date strings
            • isAfter: "2011-11-05", // only allow date strings after a specific date
            • isBefore: "2011-11-05", // only allow date strings before a specific date
            • max: 23, // only allow values <= 23
            • min: 23, // only allow values >= 23
            • isCreditCard: true, // check for valid credit card numbers
            • isEven(value) { if (parseInt(value) % 2 != 0) throw new Error('Only even values are allowed!') } // 自定义验证规则

            # 索引

            Sequelize支持在模型定义中添加索引，这些索引将在Model.sync()或期间创建sequelize.sync。

            sequelize.define('user', {}, {
  indexes: [
    // 在email上创建唯一的索引
    {
      unique: true,
      fields: ['email']
    },

    // Creates a gin index on data with the jsonb_path_ops operator
    {
      fields: ['data'],
      using: 'gin',
      operator: 'jsonb_path_ops'
    },

    // By default index name will be [table]_[fields]
    // 多列部分索引
    {
      name: 'public_by_author',
      fields: ['author', 'status'],
      where: {
        status: 'public'
      }
    },

    // 具有有序字段的BTREE索引
    {
      name: 'title_index',
      method: 'BTREE',
      fields: ['author', {attribute: 'title', collate: 'en_US', order: 'DESC', length: 5}]
    }
  ]
})

            # 模型扩展

            类型于es6中的类，Class类扩展，Class.prototype为原型扩展

            	const User = sequelize.define('user', { firstname: Sequelize.STRING })
	// 类扩展
	User.classLevelMethod = function() {
		return 'foo'
	}
	// 原型扩展
	User.prototype.instanceLevelMethod = function() {
		return 'bar'
	}
