              # 数据库链接

              const Sequelize = require('sequelize');
const sequelize = new Sequelize('database', 'username', 'password', {
	// 数据库域名，default：localhost
	host: 'localhost',

	// 数据库的端口; default: 不同数据库的默认端口，依赖于dialect
	port: 3306,

	// 数据库类型
	dialect: 'mysql'|'sqlite'|'postgres'|'mssql',

	// 附加选项，直接传递给数据库
	dialectOptions: {
		socketPath: '/Applications/MAMP/tmp/mysql/mysql.sock',
		supportBigNumbers: true,
		bigNumberStrings: true
	},

	// SQLite only
	storage: 'path/to/database.sqlite'

	// 关系数据库的协议; default: 'tcp'
	// postgres only, useful for Heroku
	protocol: null,

	// 每次Sequelize会记录某些内容时执行的函数; default: console.log
	logging: false,

	// 一个标志，用于定义是否应将空值传递给SQL查询。default: false
	omitNull: true,

	// 一个标志，用于定义是否使用本机库，目前只对postgres有效。default: false
	native: true,

	// 模型定义的默认选项
	define: {
		underscored: false
		freezeTableName: false,
		charset: 'utf8',
		dialectOptions: {
			collate: 'utf8_general_ci'
		},
		timestamps: true
	},

	// sequelize.query的默认选项
	query：{},

	// sequelize.set的默认选项
	set: {},

	// 强制模型同步更新
	sync: { force: true },

	// 运算符别名
	operatorsAliases: {
		$gt: Sequelize.Op.gt
	}

	// 时区设置
	timezone：'+/- HH：MM',

	// 连接池
	pool: {
		max: 5,                   // 最大连接数
		min: 0,                   // 最小连接数
		acquire: 30000,           // 池在抛出错误之前尝试获取连接的最长时间（以毫秒为单位）
		idle: 10000,              // 连接在释放之前可以空闲的最长时间（以毫秒为单位）
		handleDisconnects: true,  // 控制池是否应自动处理连接断开而不会抛出错误
		validate: function        // 验证连接的函数。与客户打电话。默认函数检查客户端是否为对象，并且其状态未断开
	},

	// 设置默认事务隔离级别
	isolationLevel: Transaction.ISOLATION_LEVELS.REPEATABLE_READ,

	// 在插入和更新时运行内置类型验证器，例如验证传递给整数字段的参数是否类似于整数。
	typeValidation: false
});

              # 时间戳处理

              模型定义的时候, 默认情况下，Sequelize将增加的属性createdAt和updatedAt你的模型，这样你就可以知道什么时候该数据库条目进入数据库，并当最后更新。 如果不想要createdAt和updatedAt，有两种方式可以处理

              1. 定义数据库的时候
              	// 在模型定义的默认选项中,添加以下选项
	define: {
		timestamps: false
	}
              1. 定义表结构时
              // 不需要任何时间戳(updatedAt, createdAt)
const Bar = sequelize.define('bar', { /* bla */ }, {
	timestamps: false,
})
// 时间戳不完全处理(保留updatedAt)
const Foo = sequelize.define('foo',  { /* bla */ }, {
	timestamps: true,
	createdAt: false,
	updatedAt: 'updateTimestamp',
})

              # 主键的处理

              Sequelize会假设表id默认具有主键属性，没有id字段，会自动添加，在记录创建和连表查询时，会出现意想不到的问题

              1. 定义自己的主键
              // 定义自己的主键
sequelize.define('collection', {
	uuid: {
		type: Sequelize.UUID,  // 自增类型
		primaryKey: true
	}
});
              1. 删除主键
              Model.removeAttribute('id')

              # 模型属性

              eg: User模型 模型定义中：

              • 检索基础属性值，始终使用 this.getDataValue()
              • 设置基础属性值，始终使用 this.setDataValue()

              模型方法

              • sequelize.getDialect() - 返回数据库类型
              • sequelize.getQueryInterface() - 返回QueryInterface实例
              • sequelize.define(option) -> Model - 模型定义
              • sequelize.model(modelName) -> Model - 获取模型
              • sequelize.isDefined(modelName) -> Boolean - 检查模型是否定义
              • sequelize.import(path) -> Model - 模型导入
              • sequelize.query(sql, [options={}]) -> Promise - 执行查询
              • sequelize.set(variables, options) -> Promise - 设置一个变量，设置后将会执行基于环境变量或用户变量的查询。此变量会在每次建立连接时设置，仅MySQL 适用。
              • sequelize.escape(value) -> String - 对值value进行编码并返回编码结果。
              • sequelize.createSchema() - 创建数据库 schema
              • sequelize.showAllSchemas() - 查询已定义的schema
              • sequelize.dropSchema() - 删除定义的schema
              • sequelize.dropAllSchemas() - 删除所有schema
              • sequelize.sync() - 同步模型到数据库
              • sequelize.fn() - 函数调用
              • sequelize.col() - 列对象
              • sequelize.cast() - cast函数
              • sequelize.literal() - 字面量对象
              • sequelize.and() - AND查询
              • sequelize.or() - OR查询
              • sequelize.json() - json嵌套对象
              • sequelize.where() - 指定WHERE条件
              • sequelize.transaction() - 启动事务

              # 实例属性

              eg: user实例

              • user.get({ plain: true }

              • user.reload()

              • user.increment('field', {by: 2})
              • user.increment([ 'field', 'field2' ], {by: 2})
              • user.increment({ 'field': 2, 'field2': 3 }) spread

              转化为json

              • user.toJSON()
