        # gulp简单使用

        # html压缩 -- gulp-htmlmin

        	const gulp = require('gulp');
	const htmlmin = require('gulp-htmlmin');
	
	gulp.task('minify', () => {
	return gulp.src('src/*.html')
		.pipe(htmlmin({ collapseWhitespace: true }))
		.pipe(gulp.dest('dist'));
	});

        # css文件处理 -- gulp-minify-css

        	var autoprefixer = require('gulp-autoprefixer'); // 处理css中浏览器兼容的前缀 
	var minifyCss = require('gulp-minify-css'); // css的层级压缩
	var changed = require('gulp-changed');
	var debug = require('gulp-debug');
	var rev = require('gulp-rev');// 对文件名加MD5后缀

    gulp.task('revCss', function(){
        return gulp.src(Config.css.src)
            .pipe(changed(Config.css.dist))
            .pipe(debug({title:"css"}))
            .pipe(autoprefixer())
            .pipe(minifyCss())
            .pipe(rev())
            .pipe(gulp.dest(Config.css.dist))
            .pipe(rev.manifest())
            .pipe(gulp.dest('rev/css'));
    });

        # scss文件处理 -- gulp-sass

        	var gulp = require('gulp');
	var sass = require('gulp-sass');
	
	sass.compiler = require('node-sass');
	
	gulp.task('sass', function () {
		return gulp.src('./sass/**/*.scss')
			.pipe(sass().on('error', sass.logError))
			.pipe(gulp.dest('./css'));
	});
		
	gulp.task('sass:watch', function () {
		gulp.watch('./sass/**/*.scss', ['sass']);
	});

        # 图片处理 -- gulp-imagemin

        	var imagemin = require('gulp-imagemin'); //图片压缩 
    gulp.task('images', function () {
        return gulp.src(Config.imgNotJP.src)
        .pipe(imagemin({
            optimizationLevel: 3,
            progressive: true,
            interlaced: true
        }))
		.pipe(gulp.dest(Config.imgNotJP.dist))
    });

        # 只修改已更改的文件 -- gulp-changed

        const gulp = require('gulp');
const changed = require('gulp-changed');
const ngAnnotate = require('gulp-ng-annotate'); // Just as an example
 
const SRC = 'src/*.js';
const DEST = 'dist';
 
gulp.task('default', () =>
    gulp.src(SRC)
        .pipe(changed(DEST))
        // `ngAnnotate` will only get the files that
        // changed since the last time it was run
        .pipe(ngAnnotate())
        .pipe(gulp.dest(DEST))
);

        # 删除log -- gulp-remove-logging

        var gulp_remove_logging = require("gulp-remove-logging");
gulp.task("remove_logging", function() {
  return gulp.src("src/javascripts/**/*.js")
    .pipe(
      gulp_remove_logging({
        // namespace: ['console', 'window.console']
      })
    )
    .pipe(
      gulp.dest(
        "build/javascripts/"
      )
    );
});
